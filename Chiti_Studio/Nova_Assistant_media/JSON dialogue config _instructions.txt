⚙️ 5. HOW DEV USES THIS
Example Engine
function getNode(nodeId) {
  return dialogue.nodes[nodeId];
}

function playNode(node) {
  setPersona(node.persona);
  playVideo(node.clip);

  if (node.type === "choice") {
    showOptions(node.options);
  }

  if (node.type === "input") {
    showInput(node.fields);
  }
}
