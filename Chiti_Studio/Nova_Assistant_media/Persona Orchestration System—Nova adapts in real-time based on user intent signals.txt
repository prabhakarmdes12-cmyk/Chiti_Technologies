Persona Orchestration System—Nova adapts in real-time based on user intent signals.

🧠 1. VARIANT TRIGGER LOGIC (DECISION ENGINE)

We define 3 states:

Soft Idol → Comfort Layer
Creative Director → Authority Layer
Edge / Experimental → Attention Layer
🎯 CORE SIGNALS TO TRACK

You don’t need AI initially—just structured signals:

1. Entry Behavior
Time on page
Scroll depth
Click on sphere
2. Intent Signals
Button choices
Text input keywords
Pages visited (portfolio, pricing, etc.)
3. Engagement Quality
Response speed
Drop-off points
Re-engagement
🧩 DECISION TREE (SIMPLIFIED)
IF user_first_visit THEN
    Nova = Edge_Experimental   // hook attention

IF user_clicks_sphere THEN
    Nova = Soft_Idol           // reduce friction

IF user_selects("Just exploring") THEN
    Nova = Soft_Idol

IF user_selects("I want a website") THEN
    Nova = Soft_Idol → Creative_Director (after 1 step)

IF user_selects("I’m building something new") THEN
    Nova = Creative_Director

IF user_views(pricing OR portfolio) THEN
    Nova = Creative_Director

IF user_high_intent (fast replies + specific answers) THEN
    Nova = Creative_Director

IF user_idle OR hesitant THEN
    Nova = Soft_Idol (re-engage)

IF returning_user THEN
    Nova = Creative_Director (skip fluff)

It should feel like mood change, not character replacement

🎯 C. DECISION ENGINE (CORE LOGIC)
// /lib/novaEngine.js

export function resolvePersona({ action, intent, behavior }) {
  if (behavior.firstVisit) return "edge";

  if (action === "SPHERE_CLICK") return "soft";

  if (intent === "explore") return "soft";

  if (intent === "website") return "soft";

  if (intent === "startup") return "director";

  if (behavior.visitedPricing) return "director";

  if (behavior.highIntent) return "director";

  return "soft";
}
🔄 D. HOOK IT INTO UI
import { useNovaStore } from "@/lib/novaState";
import { resolvePersona } from "@/lib/novaEngine";

function handleUserAction(payload) {
  const persona = resolvePersona(payload);
  setPersona(persona);
}
🎭 E. DYNAMIC COMPONENT SWITCH
import SoftNova from "./variants/SoftNova";
import DirectorNova from "./variants/DirectorNova";
import EdgeNova from "./variants/EdgeNova";

export default function NovaRenderer() {
  const { persona } = useNovaStore();

  switch (persona) {
    case "soft":
      return <SoftNova />;
    case "director":
      return <DirectorNova />;
    case "edge":
      return <EdgeNova />;
    default:
      return <SoftNova />;
  }
}
